<script>

    let touchStartY = 0;

    

    document.querySelector('.static-image-container').addEventListener('touchstart', function(event) {

        touchStartY = event.touches[0].clientY; // Anfangs-Y-Position speichern

    });


    document.querySelector('.static-image-container').addEventListener('touchmove', function(event) {

        let touchEndY = event.touches[0].clientY;

        let touchDifference = touchEndY - touchStartY;


        if (Math.abs(touchDifference) < 0) {

            event.stopPropagation(); // Wenn der Unterschied klein ist, wird das Scrollen verhindert

        } else {

            event.preventDefault(); // Scrollen der Seite erlauben, nicht des Containers

        }

    });

</script>
